>>> txt = sc.textFile("/home/students/astatochek/file.txt")
>>> rdd = txt.flatMap(lambda line: line.split(" ")).map(lambda word: (word, 1)).reduceByKey(lambda x, y: x + y)
>>> out = rdd.map(lambda data: ",".join(f"{item}" for item in data))
>>> out.saveAsTextFile("/home/students/astatochek/out")
